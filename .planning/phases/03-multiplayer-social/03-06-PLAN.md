    **Create `apps/server/Services/ChatFilter.cs`:**
    Server-side word filter. Since the chat messages go through a C# API endpoint before being published to AppSync (server-side publishing ensures integrity), implement a basic filter in C#:
    ```csharp
    public class ChatFilter
    {
        // A curated list of profanity words — ported from leo-profanity's dictionary
        // Start with a reasonable default set (~200 words)
        private static readonly HashSet<string> _blockedWords = new(StringComparer.OrdinalIgnoreCase)
        {
            // ... populate from leo-profanity's word list or a standard English profanity set
        };

        public string Filter(string message)
        {
            // Split by whitespace, replace blocked words with asterisks
            // Handle common obfuscation: l33t speak basics (a→@, e→3, etc.)
            var words = message.Split(' ');
            for (int i = 0; i < words.Length; i++)
            {
                var normalized = NormalizeLeetSpeak(words[i].ToLower());
                if (_blockedWords.Contains(normalized))
                    words[i] = new string('*', words[i].Length);
            }
            return string.Join(' ', words);
        }

        private static string NormalizeLeetSpeak(string word)
        {
            return word.Replace('@', 'a').Replace('3', 'e').Replace('1', 'i')
                       .Replace('0', 'o').Replace('$', 's').Replace('5', 's');
        }
    }
    ```
    The word list can be loaded from a text file or embedded as a resource. Start with a well-known English profanity list of ~200 words.

      - Creates a `PlayerReport` row in the DB (logged for future moderation review)
      - Returns 200 `{ success: true }`

    **Extend `apps/client/src/lib/appsync.ts`:**
    Add chat channel subscription:
    ```typescript
    export async function subscribeToChatChannel(
      channelId: string,
      onMessage: (msg: ChatMessage) => void,
    ): Promise<(() => void) | null> {
      if (!configured) return null;
      try {
        const channel = await events.connect(`/game/${channelId}/chat`);
        channel.subscribe({
          next: (data: any) => {
            const parsed = typeof data.event === 'string' ? JSON.parse(data.event) : data.event;
            onMessage(parsed as ChatMessage);
          },
          error: console.error,
        });
        return () => channel.close();
      } catch { return null; }
    }

    export interface ChatMessage {
      userId: string;
      username: string;
      message: string;
      timestamp: string;
    }
    ```

    <script lang="ts">
      import { onMount, onDestroy } from 'svelte';
      import { browser } from '$app/environment';
      import { configureAppSync, subscribeToChatChannel, type ChatMessage } from '$lib/appsync';

      // Props
      let { channelId, currentUsername }: { channelId: string; currentUsername: string } = $props();

      let messages = $state<ChatMessage[]>([]);
      let inputText = $state('');
      let cleanup: (() => void) | null = null;
      let chatContainer: HTMLDivElement;

      onMount(async () => {
        if (!browser) return;
        configureAppSync();
        cleanup = await subscribeToChatChannel(channelId, (msg) => {
          messages = [...messages, msg];
          // Auto-scroll to bottom
          requestAnimationFrame(() => chatContainer?.scrollTo(0, chatContainer.scrollHeight));
        });
      });

      onDestroy(() => cleanup?.());

      async function sendMessage() {
        if (!inputText.trim()) return;
        const token = await getApiToken(); // from auth helper
        await fetch(`${API_BASE}/chat/${channelId}/send`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${token}` },
          body: JSON.stringify({ message: inputText.trim() }),
        });
        inputText = '';
      }
    </script>
